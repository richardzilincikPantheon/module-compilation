21import os 

22import shutil 

23import subprocess 

24import unittest 

25 

26from ietf_modules_extraction import yang_version_1_1 as yv11 

27 

28 

29class TestYangVersion11(unittest.TestCase): 

30 @classmethod 

31 def setUpClass(cls): 

32 cls.resource_path = os.path.join(os.environ['VIRTUAL_ENV'], 'tests/resources/yang_version_1_1') 

33 cls.script_path = os.path.join(os.environ['VIRTUAL_ENV'], 'ietf_modules_extraction/yang_version_1_1.py') 

34 cls.src = os.path.join(cls.resource_path, 'YANG') 

35 cls.dst = os.path.join(cls.resource_path, 'YANG-v11') 

36 

37 def setUp(self): 

38 shutil.rmtree(self.dst, ignore_errors=True) 

39 

40 def test_yang_version_1_1(self): 

41 """Find and copy the yang files that contain 'yang-version 1.1' string.""" 

42 result = yv11.find_v11_models(self.src, self.dst, 1) 

43 self.assertNotEqual(result, []) 

44 self.assertIn('test.yang', result) 

45 self.assertNotIn('test2.yang', result) 

46 

47 v1_files = os.listdir(self.dst) 

48 self.assertNotEqual(v1_files, []) 

49 self.assertIn('test.yang', v1_files) 

50 

51 def test_yang_version_1_1_src_not_exists(self): 

52 """Test the case when the src directory does not exist.""" 

53 result = yv11.find_v11_models('', self.dst) 

54 self.assertEqual(result, []) 

55 

56 def test_yang_version_1_1_from_console(self): 

57 """Run the script from the console by passing the arguments.""" 

58 bash_command = f'python {self.script_path} --srcpath {self.src} --dstpath {self.dst} --debug 1' 

59 subprocess.run(bash_command, shell=True, capture_output=True, check=False).stdout.decode() 

60 

61 v1_files = os.listdir(self.dst) 

62 self.assertNotEqual(v1_files, []) 

63 self.assertIn('test.yang', v1_files)

65 

66if __name__ == '__main__': 

67 unittest.main()
